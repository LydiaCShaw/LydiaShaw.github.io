%% 此段代码为提取颜色特征所用代码
%颜色直方图
%运行时请先替换图片存储路径
%尚麟静
%20155467
%基于最小生成树的宫颈癌病理图像聚类研究
%% 
close all;
clear all;
clc;

img_original = imread('D:\宇宙第一ADC\____学习\毕业设计\新建文件夹\新建文件夹\he106-5.png');%读入图像
figure,imshow(img_original),title('原始图像');    %显示原图像

R=img_original(:,:,1);%把RGB各个分量提出
G=img_original(:,:,2);
B=img_original(:,:,3);
figure;%画出直方图
subplot(1,3,1),imhist(R),title('红色');
subplot(1,3,2),imhist(G),title('绿色');
subplot(1,3,3),imhist(B),title('蓝色');

siz=size(img_original);
I1=reshape(img_original,siz(1)*siz(2),siz(3));  % 每个颜色通道变为一列
I1=double(I1);
[N,X]=hist(I1, [0:1:255]);    % 如果需要小矩形宽一点，划分区域少点，可以把步长改大，比如0:5:255
figure;
bar(X,N(:,[3 2 1]));    % 柱形图，用N(:,[3 2 1])是因为默认绘图的时候采用的颜色顺序为b,g,r,c,m,y,k，跟图片的rgb顺序正好相反，所以把图片列的顺序倒过来，让图片颜色通道跟绘制时的颜色一致
xlim([0 255])
hold on
plot(X,N(:,[3 2 1]));    % 上边界轮廓
hold off
