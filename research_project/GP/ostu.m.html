%% 此段为大津阈值分割尝试以及图像分割形态学处理代码
%运行前请先替换图片存储路径
%尚麟静
%基于最小生成树的宫颈癌病理图像聚类研究
%% 
clc
clear
close all
I=imread('D:\宇宙第一ADC\____学习\毕业设计\新建文件夹\新建文件夹\he106-1.png');
%读取图片
I1=rgb2gray(I); %灰度化
T=graythresh(I1);
I2=im2bw(I1,T);   %二值化
figure,imshow(I2,[]); 
title('大津阈值二值化')
%% 形态学处理
%结构元素
se=strel('disk',2');
se2=strel('disk',3');
se3=strel('disk',4');
%图像膨胀
I3=imdilate(I2, se);
figure,imshow(I3);
%图像腐蚀
I4=imerode(I3,se3);
figure,imshow(I4);
%图像填充 
I42=~I4;
%I4=imfill(I3, 'holes');
%figure,imshow(I4)
figure,imshow(I42)
%骨架化及细化
BW = bwmorph(I42,'skel',Inf); 
BW = bwmorph(BW,'thin',Inf);
figure,imshow(BW);
imwrite(I42,'106-1.png');
